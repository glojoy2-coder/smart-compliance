<script>
  // ─────────────────────────────────────────────
  //  Get customer token from URL or localStorage
  // ─────────────────────────────────────────────
  function getToken() {
    const params = new URLSearchParams(window.location.search);
    return params.get('token') || localStorage.getItem('sc_token') || '';
  }

  // ─────────────────────────────────────────────
  //  Subscribe — call /api/checkout, redirect
  // ─────────────────────────────────────────────
  async function subscribe(plan) {
    const token = getToken();
    const overlay  = document.getElementById('loadingOverlay');
    const errBanner = document.getElementById('errorBanner');
    errBanner.classList.remove('show');
    overlay.classList.add('show');

    try {
      const res = await fetch('/api/checkout', {
        method: 'POST',
        headers: {
          'Content-Type': 'application/json',
          'Authorization': 'Bearer ' + token
        },
        body: JSON.stringify({ plan })
      });

      const data = await res.json();

      if (!res.ok) throw new Error(data.error || 'Checkout failed');

      // Redirect to Stripe hosted checkout
      window.location.href = data.url;

    } catch (err) {
      overlay.classList.remove('show');
      errBanner.textContent = err.message || 'Something went wrong. Please try again.';
      errBanner.classList.add('show');
    }
  }
</script>
